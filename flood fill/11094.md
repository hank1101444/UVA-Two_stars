#include <iostream>
using namespace std;
#include <cstring>
#include <string>
int M, N, X, Y, sum, mx;
char land, tp;
char graph[25][25];
int check[25][25];
int dir[4][2] = {{0, -1}, {0,1}, {1,0}, {-1, 0} };
string s;
void dfs(int x, int y) {
	if(graph[x][y] == land && check[x][y] == 0){
		sum += 1;
		check[x][y] = 1;
		for (int i = 0; i < 4; ++i) {
			if (x + dir[i][0] < 0 || x + dir[i][0] == M)
				continue;
			dfs((x + dir[i][0]) % N, y + dir[i][1]);
		}
	}
}

int main() {
	while (cin >> M >> N) {
		mx = 0;
		memset(check, 0, sizeof(check));
		for (int i = 0; i < M; ++i) {
			cin >> s;
			for (int j = 0; j < s.size(); ++j) {
				graph[i][j] = s[j];
			}
		}
		// king
		cin >> X >> Y;
		land = graph[X][Y];
		dfs(X, Y);	// first time no calculate
		for (int i = 0; i < M; ++i) {
			for (int j = 0; j < N; ++j) {
				if (graph[i][j] == land && check[i][j] != 1) {
					sum = 0;
					dfs(i, j);
					if (mx < sum) mx = sum;
				}

			}
		}
		cout << mx << '\n';
	}

	return 0;
}
