```cpp

#include <iostream>
using namespace std;
#include <cmath>

double v, v0, ans = 0;
double mx;
void solve() {
	for (double i = 1; v / i > v0; ++i) {
		double tp = v / i;
		if (tp <= v0)
			return;
		else {
			double tmp = 0.3 * sqrt(tp - v0) * i;
			if (mx < tmp) {
				mx = tmp;
				// 最大更改時, ans 一起變動 就算後面有
				ans = i;
			}
			// remark
			// 如果最大的出現兩次ans 為0, 卡在擔心後面可能有較大值
			else if (mx == tmp)
				ans = 0;
		}
	
	}
}



int main() {
	while (cin >> v >> v0) {
		if (!v && !v0)
			break;
		mx = 0;
		ans = 0;
		solve();
		cout << ans << endl;
	}
	return 0;
}
```
