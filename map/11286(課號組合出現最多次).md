## 利用 string 來當key 
```cpp
#include <iostream>
using namespace std;
#include<string>
#include <algorithm>
#include <map>

// 因為我不希望課號裡面的數字也被sort 所以創一個容器來存
string course[5];

int main() {
	// 因為最後想用stirng 來當key 所以一開始就讀string
	int n;
	string key;
	map<string, int> m;
	while (cin >> n && n != 0) {
		int count = 0, max = 0;
		m.clear();
		key.clear();
		while (n--) {
			for (int i = 0; i < 5; ++i) {
				//int a;		// modified
				//cin >> a;
				string s;
				cin >> s;
				course[i] = s;
			}
			// sort (reference, end reference)
			//sort(&course[0], &course[0] + sizeof(course) / sizeof(int));
			sort(course, course + sizeof(course) / sizeof(course[0]));
			string compose;
			for (int i = 0; i < 5; ++i)
				compose += course[i];
			if (m.find(compose) != m.end())
				++m[compose];
			else
				m[compose] = 1;
			if (m[compose] > max)
				max = m[compose];
		}
		for (map<string, int>::iterator it = m.begin(); it != m.end(); ++it) {
			if(it->second == max)
				count += it->second;
		}
		cout << count << endl;
	}
	return 0;
}
```
