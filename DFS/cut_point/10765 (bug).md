#include<iostream>
using namespace std;
#include <vector>
#include <string>
#include <map>
#include <algorithm>
vector <vector <int>> graph;
int node_num, m, times, cs;// 0 ~
int adj1, adj2;// 0~



struct Node {
	int id = -1;
	int d=0;
	int low=0;
	int child_num=0;
	int par = -1;
	bool cut = false;
};
vector <pair<int, int>> ans;
vector <Node> vertex;
bool cmp(pair<int, int> a, pair<int,int> b) {
	if (a.second == b.second)
		return a.first < b.first;
	return a.second > b.second;
}


// check current u == cut Point
void cutPoint (int u) {
	++times;
	vertex[u].d = times;
	vertex[u].low = times;
	for (int i = 0; i < graph[u].size(); ++i) {
		int v = graph[u][i];
		if (vertex[v].d == 0) {
			vertex[v].par = u;
			// child 數目就是連通的聯通圖數目，
			vertex[u].child_num++;
			cutPoint(v);
			vertex[u].low = min(vertex[v].low, vertex[u].low);
			if (vertex[u].par != -1) {
				if (vertex[v].low >= vertex[u].d) {
					// is a cutPoint
					vertex[u].cut = true;
					ans[u].second++;
				}
			}
		}
		else if (vertex[u].par != v) {
			// backedge
			vertex[u].low = min(vertex[u].low, vertex[v].d);
		}
	}
	if (vertex[u].par == -1) {
		if (vertex[u].child_num > 1) {
			vertex[u].cut = true;
			ans[u].second = vertex[u].child_num;
		}
	}
}

int main() {
	// cout m max bomb
	while (cin >> node_num >> m && node_num && m) {
		vertex.assign(node_num, Node());
		graph.assign(node_num, vector <int>());
		ans.assign(node_num, { 0,0 });
		times = 0;
		while (cin >> adj1 >> adj2 && adj1 != -1 && adj2 != -1) {
			graph[adj1].push_back(adj2);
			graph[adj2].push_back(adj1);
		}
		for (int i = 0; i < node_num; ++i) {
			
			ans[i] = (make_pair(i, 1));
		}

		for (int i = 0; i < node_num; ++i) {
			// output need station id
			if (vertex[i].d == 0)
				cutPoint(i);
		}
		sort(ans.begin(), ans.end(), cmp);
		for (int i = 0; i < m; ++i) {
			cout << ans[i].first << ' ' << ans[i].second << '\n';
		}
	}
	// 當所有cutPoint 都輸出完後 開始炸一般點，一般點消失則剩下的聯通圖只有一塊
	return 0;
}
