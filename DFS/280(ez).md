/*
題目要求：給定單向圖及起始位置，輸出不會經過的點
輸入 : nodes 表示graph 有幾個點，且方便我找前幾個node 即可
有向圖由 n 個頂點表示，其中 1 ≤ n ≤ 100，連續(consecutively)編號為 1
輸出 ：單向不會走到的點
*/

#include <iostream>
using namespace std;
#include <cstring>
#include <vector>
int nodes, ckt;
int n1, ntp;
int graph[200][200];
int record[200];
vector <int> v;

void dfs(int curr) {
	// question says that 1 <= i <= n
	for (int i = 1; i <= nodes; ++i) {
		// 已經走過的不需要再進入dfs
		if (graph[curr][i] == 1 && record[i] == 0) {
			record[i] = 1;
			dfs(i);
		}
	}
}

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);

	// nodes 個數
	while (cin >> nodes && nodes) {
		memset(graph, 0, sizeof(graph));
		while (cin >> n1 && n1 != 0) {
			while (cin >> ntp && ntp != 0) {
				graph[n1][ntp] = 1;
			}
		}
		//check
		cin >> ckt;
		while (ckt--) {
			memset(record, 0, sizeof(record));
			// using vector because output last no need space
			v.clear();
			cin >> ntp;
			//record[ntp] = 1;
			dfs(ntp);
			// 要輸出沒通過的點
			for (int i = 1; i <= nodes; ++i) {
				if (record[i] == 0)
					v.push_back(i);
			}
			cout << v.size();
			for (int i = 0; i < v.size(); ++i) {
				cout << ' ' << v[i];
			}
			cout << '\n';
		}
	}
	return 0;
}
