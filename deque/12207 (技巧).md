#include <iostream>
using namespace std;
#include <deque>

// deque用在要traversal 且需要常push pop 前後時


int main() {

	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cout.tie(nullptr);

	int p, c, count = 1;
	char com;
	while (cin >> p >> c && (p != 0 && c != 0)) {
		cout << "Case " << count++ << ":\n";
		deque <int> dq;
		// 若 C 的值小於 P 的值則我們一開始只需要推入 C 個序號即可，因為後面的序號除非被往前移動(E)，否則絕對不會使用到。
		for (int i = 1; i <= min(p, c); ++i)
			dq.push_back(i);
		for (int i = 0; i < c; ++i) {
			cin >> com;
			if (com == 'N') {
				int a = dq.front();
				dq.push_back(a);
				cout << dq.front() << '\n';
				dq.pop_front();
			}
			else {
				int get;
				cin >> get;
				for (deque <int> ::iterator it = dq.begin(); it != dq.end(); ++it) {
					// 如果deque 裡面有則要刪除
					if (*it == get) {
						// error??? 先push 再 erase 會出錯
						dq.erase(it);
						//dq.push_front(get);
						break;
					}
				}
				// 此時deque 沒此元素 將此加在最前面
				dq.push_front(get);
			}
		}
	}




	return 0;
}
