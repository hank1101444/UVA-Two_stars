## 注意裝箱或卸貨時 算是一個動作 放到哪裡不算一個動作     
## 注意empty問題
```cpp
#include <iostream>
using namespace std;
#include <vector>
#include <queue>
#include <stack>

// 注意裝箱或卸貨時 算是一個動作 放到哪裡不算一個動作
int main() {
	int n;
	cin >> n;
	while (n--) {
		stack <int> carrier;
		int N, S, Q, tp, total = 0;
		vector < queue <int>> v;
		cin >> N >> S >> Q;
		for (int i = 0; i < N; ++i) {
			int numbers;
			cin >> numbers;
			queue <int> tp_q;
			for (int j = 0; j < numbers; ++j) {
				cin >> tp;
				// tp要記得減一 因為我是從0開始放不是從
				tp--;
				tp_q.push(tp);
			}
			v.push_back(tp_q);
		}
		int curr = 0;
		while (1) {
			// 先卸貨
			// 不管怎樣 我都會先卸貨 最後再看是要push 進去還是直接丟工廠
			while (!carrier.empty() && (v[curr].size() < Q || carrier.top() == curr)) {
				tp = carrier.top();
				carrier.pop();
				total += 1;
				// carrier 可能已經沒了
				//if (carrier.top() != curr) {
				if (tp != curr) {
					v[curr].push(tp);
				}
			}
			//再裝箱
			while (!v[curr].empty() && carrier.size() < S) {
				tp = v[curr].front();
				v[curr].pop();
				carrier.push(tp);
				total += 1;
			}
			bool check = false;
			for (int i = 0; i < v.size(); ++i) {
				if (v[i].size() > 0) {
					check = true;
					break;
				}
			}
			if (!check && !carrier.size())break;

			// 到下一站
			total += 2;
			curr += 1;
			curr %= v.size();
		}
		cout << total << endl;
	}

	return 0;
}
```
