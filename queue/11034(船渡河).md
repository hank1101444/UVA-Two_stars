## 題目說從 left 開始
## 就算一邊空了還是要回來載
```cpp
#include <iostream>
using namespace std;
#include <queue>
#include <string>
#include <utility>
int main() {
	int n;
	cin >> n;
	while (n--) {
		int l, m, tp, count = 0;
		cin >> l >> m;
		l *= 100;
		queue <int>q[2];
		string s;
		for (int i = 0; i < m; ++i) {
			cin >> tp >> s;
			if (s == "left")
				q[0].push(tp);
			else
				q[1].push(tp);
		}
		// 題目說從left 開始
		int rl = 0;
		while(!q[0].empty() || !q[1].empty()) {
			if (!q[rl].empty()) {
				tp = 0;
				while (!q[rl].empty() && tp + q[rl].front() <= l) {
					tp += q[rl].front();
					q[rl].pop();
				}
			}
			// 就算一邊空了還是要回來載
			++count;
			rl += 1;
			rl %= 2;
			// 如果兩邊都沒了就離開
		}
		cout << count << endl;
	}

	return 0;
}
```
