## WA 算出來得極小值可能為負，可是最小面積只能是0 所以回去看讓面積為0的方法就是讓x = 最短邊 / 2
## 可能會有虛根問題 回到題目尋找限制
```cpp
#include <iostream>
using namespace std;
#include <cmath>
#include <iomanip>
#include <cstring>
double mx, mn, ans[2];

int main() {
	double l, w;
	while (cin >> l >> w) {
		//memset(mx, 0, sizeof(mx));
		//memset(mn, 0, sizeof(mn));
		memset(ans, 0, sizeof(ans));
		ans[0] = (l + w + sqrt(pow(l + w, 2) - 3 * w * l)) / 6;
		ans[1] = (l + w - sqrt(pow(l + w, 2) - 3 * w * l)) / 6;
		if ((ans[0] * 24 - 4 * l - 4 * w) > 0) {
			mn = ans[0];
			mx = ans[1];
		}
		else {
			mn = ans[1]; 
			mx = ans[0];
		}
		cout << fixed << setprecision(3) << mx << ' ' << "0.000" << ' ' << fixed << setprecision(3) << mn << endl;
	}
	return 0;
}
```

## AC
```cpp
#include <iostream>
using namespace std;
#include <cmath>
#include <iomanip>
#include <cstring>
double mx, mn, ans[2];

int main() {
	double l, w;
	while (cin >> l >> w) {
		// mx
		memset(ans, 0, sizeof(ans));
		ans[0] = (l + w + sqrt(pow(l + w, 2) - 3 * w * l)) / 6;
		ans[1] = (l + w - sqrt(pow(l + w, 2) - 3 * w * l)) / 6;
		for (int i = 0; i < 2; ++i) {
			if ((ans[0] * 24 - 4 * l - 4 * w) > 0) 
				mx = ans[i];
		}
		// mn 發生在原點 (x(hight) = 0) or 面積為0 (x max 為最短邊的 一半)
		mn = (l > w ? w : l) / 2;
		// 必過原點
		cout << fixed << setprecision(3) << mx << ' ' << "0.000" << ' ' << fixed << setprecision(3) << mn << endl;
	}
	return 0;
}
```
